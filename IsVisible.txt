
bool USkeletalMeshComponent::IsVisible()
{
    auto Seconds = request->read<double>(reinterpret_cast<uintptr_t>(Cheat::Cache::World) + 0x158 + 0x10);
    auto LastRenderTime = request->read<float>(reinterpret_cast<uintptr_t>(this) + 0x32C);
    return Seconds - LastRenderTime <= 0.06f;
